// GAS 편집기에 아래 코드를 붙여넣으세요 function doGet(e) { var action = e.parameter.action; if (action === 'getSetupData') { return createJsonResponse(getSetupData()); } return ContentService.createTextOutput("Hello, API is running"); } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("점검결과"); var timestamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm:ss"); var rows = data.map(function(r) { return [timestamp, r.worker, r.period, r.line, r.no, r.itemName, r.result, r.remark1, r.remark2]; }); sheet.getRange(sheet.getLastRow() + 1, 1, rows.length, rows[0].length).setValues(rows); return createJsonResponse({success: true}); } catch(e) { return createJsonResponse({success: false, error: e.toString()}); } } // CORS 해결을 위한 도우미 함수 function createJsonResponse(data) { var output = ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); output.setHeader("Access-Control-Allow-Origin", "*"); output.setHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS"); return output; } // 기존 getSetupData 함수는 그대로 유지 function getSetupData() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var lines = ss.getSheetByName("설정_라인").getDataRange().getValues().slice(1); var items = ss.getSheetByName("설정_점검항목").getDataRange().getValues().slice(1); var workers = ss.getSheetByName("설정_점검자").getDataRange().getValues().slice(1).map(r => r[0]); return { lines: lines.map(r => ({period: r[0], lineName: r[1]})), items: items.map(r => ({lineName: r[0], no: r[1], itemName: r[2]})), workers: workers.filter(String) }; }
